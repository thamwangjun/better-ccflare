```typescript
import { describe, expect, it } from "bun:test";
import { OpenRouterProvider } from "../provider";

describe("OpenRouterProvider.transformRequestBody", () => {
	it("injects cache_control per-block at tools, system, and last assistant turn", async () => {
		const provider = new OpenRouterProvider();
		const body = {
			model: "anthropic/claude-sonnet-4-6",
			tools: [{ name: "get_weather", description: "Get current weather" }],
			system: "You are a helpful assistant",
			messages: [
				{ role: "user", content: "What is the weather?" },
				{ role: "assistant", content: "Let me check that for you." },
			],
			max_tokens: 10,
		};
		const request = new Request("https://openrouter.ai/api/v1/messages", {
			method: "POST",
			headers: { "content-type": "application/json" },
			body: JSON.stringify(body),
		});

		const transformed = await provider.transformRequestBody(request);
		const result = await transformed.json();

		// Breakpoint 1: last tool (D-02, D-06)
		expect(result.tools[result.tools.length - 1].cache_control).toEqual({
			type: "ephemeral",
		});

		// Breakpoint 2: system string converted to array with cache_control on last block (D-02, D-06)
		expect(result.system).toEqual([
			{
				type: "text",
				text: "You are a helpful assistant",
				cache_control: { type: "ephemeral" },
			},
		]);

		// Breakpoint 3: last assistant turn content converted to array with cache_control (D-02, D-06)
		const lastAssistant = [...result.messages]
			.reverse()
			.find((m: any) => m.role === "assistant");
		expect(Array.isArray(lastAssistant.content)).toBe(true);
		expect(
			lastAssistant.content[lastAssistant.content.length - 1].cache_control,
		).toEqual({ type: "ephemeral" });

		// No top-level cache_control key (D-05: per-block only)
		expect(result.cache_control).toBeUndefined();
	});

	it("preserves existing body fields and has no top-level cache_control", async () => {
		const provider = new OpenRouterProvider();
		const body = {
			model: "anthropic/claude-sonnet-4-6",
			messages: [{ role: "user", content: "hello" }],
			max_tokens: 50,
			stream: true,
		};
		const request = new Request("https://openrouter.ai/api/v1/messages", {
			method: "POST",
			headers: { "content-type": "application/json" },
			body: JSON.stringify(body),
		});

		const transformed = await provider.transformRequestBody(request);
		const result = await transformed.json();

		expect(result.model).toBe("anthropic/claude-sonnet-4-6");
		expect(result.max_tokens).toBe(50);
		expect(result.stream).toBe(true);
		// Per D-05: no top-level cache_control; per-block only
		expect(result.cache_control).toBeUndefined();
	});
});

describe("OpenRouterProvider.extractUsageInfo", () => {
	it("extracts cache_write_tokens from non-streaming OpenRouter response (regression guard)", async () => {
		// PATCH-02 / D-08 / D-09: Regression guard — happy path only.
		// If the extractUsageInfo override is removed from OpenRouterProvider,
		// the inherited BaseAnthropicCompatibleProvider.extractUsageInfo() reads
		// cache_creation_input_tokens (absent below), returning 0. This test MUST fail
		// if the patch is removed.
		const provider = new OpenRouterProvider();
		const responseBody = {
			model: "anthropic/claude-sonnet-4-6",
			usage: {
				prompt_tokens: 100,
				completion_tokens: 10,
				total_tokens: 110,
				prompt_tokens_details: {
					cached_tokens: 0,
					cache_write_tokens: 50,
				},
			},
		};
		const response = new Response(JSON.stringify(responseBody), {
			status: 200,
			headers: { "content-type": "application/json" },
		});

		const usage = await provider.extractUsageInfo(response);

		expect(usage).not.toBeNull();
		expect(usage!.cacheCreationInputTokens).toBe(50);
	});
});
```
