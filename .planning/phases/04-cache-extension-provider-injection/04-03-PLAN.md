// FORK PATCH: 3-breakpoint cache_control injection (tools, system, last assistant turn)
override async transformRequestBody(
  request: Request,
  account?: Account,
): Promise<Request> {
  const mapped = await super.transformRequestBody(request, account);
  try {
    const body = await mapped.clone().json();
    if (body && typeof body === "object") {
      // Breakpoint 1: last tool (overwrites — Phase 4 must make non-destructive)
      if (Array.isArray(body.tools) && body.tools.length > 0) {
        const lastTool = body.tools[body.tools.length - 1];
        if (lastTool && typeof lastTool === "object") {
          (lastTool as any).cache_control = { type: "ephemeral" }; // <-- unconditional overwrite
        }
      }
      // Breakpoint 2: system (string or array — overwrites — Phase 4 must make non-destructive)
      if (typeof body.system === "string" && body.system.length > 0) {
        body.system = [{ type: "text", text: body.system, cache_control: { type: "ephemeral" } }];
      } else if (Array.isArray(body.system) && body.system.length > 0) {
        const lastBlock = body.system[body.system.length - 1];
        if (lastBlock && typeof lastBlock === "object") {
          (lastBlock as any).cache_control = { type: "ephemeral" }; // <-- unconditional overwrite
        }
      }
      // Breakpoint 3: last assistant turn (overwrites — Phase 4 must make non-destructive)
      if (Array.isArray(body.messages)) {
        const lastAssistant = [...body.messages].reverse().find((m: any) => m.role === "assistant");
        if (lastAssistant) {
          if (Array.isArray(lastAssistant.content) && lastAssistant.content.length > 0) {
            const lastBlock = lastAssistant.content[lastAssistant.content.length - 1];
            if (lastBlock && typeof lastBlock === "object") {
              (lastBlock as any).cache_control = { type: "ephemeral" }; // <-- unconditional overwrite
            }
          } else if (typeof lastAssistant.content === "string" && lastAssistant.content.length > 0) {
            lastAssistant.content = [{ type: "text", text: lastAssistant.content, cache_control: { type: "ephemeral" } }];
          }
        }
      }
      log.debug("Injected cache_control breakpoints into OpenRouter request");
      return new Request(mapped.url, { method: mapped.method, headers: mapped.headers, body: JSON.stringify(body) });
    }
  } catch (error) {
    log.debug("Failed to inject cache_control:", error);
  }
  return mapped;
}

| account.openrouter_provider_preference TEXT → parsed object | Stored data could be malformed; parse failures must not break the proxy request path |
